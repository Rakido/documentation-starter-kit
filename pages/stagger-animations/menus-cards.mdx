## Menu Items

Animation de menu avec stagger :

```html
<ul 
  data-stagger-reveal 
  data-stagger="0.1"
  data-direction="y"
  data-distance="15"
>
  <li>Accueil</li>
  <li>Services</li>
  <li>Portfolio</li>
  <li>Contact</li>
</ul>
```

<NxStaggerAnimation 
  data-stagger-reveal 
  data-stagger="0.1"
  data-direction="y"
  data-distance="15"
  data-easing="power4.out"
  style={{
    display: 'flex',
    gap: '1rem',
  }}
>
  {['Accueil', 'Services', 'Portfolio', 'Contact'].map((item) => (
    <div 
      key={item}
      data-stagger-item
      className="nx-bg-gray-700 nx-p-4 nx-rounded-lg nx-mb-4 nx-cursor-pointer hover:nx-bg-gray-600"
    >
      {item}
    </div>
  ))}
</NxStaggerAnimation>
